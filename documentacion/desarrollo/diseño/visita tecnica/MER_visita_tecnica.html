Table terreno

IdxField NameData TypeDescription
* identifica_p varchar(30) Identificación o codigo propio del predio (registro catastral)

  cedula_ppal_p varchar(2) Indica si la cédula catastral es la principal o no. (CISA-SIGA)
id_catastro_p varchar(30) Código asignado al Bien Inmueble por la Oficina de Catastro Municipal
  nupre_p varchar(6) Este campo almacena el número predial registral a implementarse para el catastro multipropósito, código de seis digitos en base 36
  dv_p varchar(1) campo que almacena el digito de verificación del nupre.
codigonal_p varchar(30) Número que identifica de forma única a cada predio a nivel nacional.
  codigounico_p varchar(30) Número que identifica de forma única a cada predio.
id_proyecto_p integer Llave foranea para el control y asociación unica del predio con el proyecto urbanístico al que pertenece
clase_inmueble_p integer Indica la Clase de Bien Inmueble respecto a si es Tereno, Terreno y Construcción, o Sometido al regimen de prop. horizontal.
id_cb_fk integer Llave foránea que relaciona esta tabla con la llave principal de la tabla Calidad de Bien.
id_tb_fk integer Llave foránea que relaciona esta tabla con la llave principal de la tabla Tipo de Bien.
id_tu_fk integer Llave foránea que relaciona esta tabla con la tabla uso predio.
  direccion_p varchar(90) Dirección del predio
  direccioncatastro_p varchar(80) Dirección del predio tal como aparece en catastro.
  zona_p varchar(7) Zona en la que está ubicado el predio: Urbana o Rural.
id_barrio varchar(254) Llave foránea que relaciona esta tabla con la tabla Barrios. A través de la tabla barrios se puede saber la comuna y el estrato de un predio.
  pais_p varchar(3) Código segun el DANE del pais del Bien Inmueble.
ciudad_p varchar(4) Código según el DANE de la Ciudad del Bien Inmueble.
  lind_norte_p varchar(400) Limite norte del predio
  lind_sur_p varchar(400) Limite sur del predio
  lind_este_p varchar(400) Limite este del predio
  lind_oeste_p varchar(400) Limite oeste del predio
  lind_adic_p varchar(600) Describe los linderos adicionales que para algunos predios, figuren registrados en la escritura de cesión
  matricula_ppal_p varchar(2) Indica si la Matricula Inmobiliaria es la Principal o No.
mat_inmob_p varchar(25) Numero de matricula inmobiliaria del predio. Número de registro, asignado al certificado de tradición, que determina la propiedad de un bien inmueble y su situación jurídica
id_madq_fk integer Llave foránea que relaciona esta tabla con la llave principal de la tabla Modo_adq (Modo de Adquisición del predio)
  nombre_areacedida_p varchar(90) Palabras designada por el urbanizador para referirse a cada una de las cesiones, con el fin de ser diferenciadas entre ellas.
nit_cede_fk integer Llave foránea que relaciona esta tabla con la tabla cedente, para identificar a la persona/entidad que cedió el predio.
  derecho_p float8 Porcentaje de propiedad que tiene el municipio sobre el predio.
  afecta_pot_p varchar(2) Indica si el Bien Inmueble tiene Afectaciones POT
  asegurado_p varchar(2) Indica si el Bien Inmueble esta Asegurado o No.
  suscep_vta_p varchar(2) Indica si el predio es susceptible para la venta.
id_depen_fk integer Llave foránea que relaciona esta tabla con la Tabla Dependencia.
  nombrecomun_p varchar(100) Nombre común del predio, tal como se le conoce entre los habitantes del sector o comunidad caleña.
  area_cesion_p float8 Medida de superficie, que El urbanizador cedió como obligación urbanística y está consignada en la escritura y plano de cesión del predio
  area_actual_p float8 Medida de superficie, obtenida a partir del levantamiento topográfico y los cálculos realizados, según composición actual del terreno
  area_sicat_p float8 Área según catastro.
  area_terreno_p float8 Área total del terreno. Para cálculo del avalúo catastral componente terreno (excepto zonas verdes que No se avalúan).
  fecha_estudio_titulo_p date Indica la fecha en la que se realizó el estudio jurídico al bien inmueble
  num_activofijo_p varchar(20) Registro contable, que determina el alta de los bienes inmuebles en los Estados Financieros del Municipio.
  codigo_zhg_p varchar(12) Código zona geográfica.
cuenta_terreno_p varchar(15) llave foránea que relaciona esta tabla con la tabla cuenta, indica el número de la cuenta del terreno (SAP)
propietario_antes_p integer Indica el Nombre del Propietario anterior del Bien Inmueble.
  actualiza_sap date Este Campo almacena la fecha en la que el registro fue actualizado o creado en el sistema SAP.
  impto_predial_p varchar(20) Indica el estado del Impuesto Predial del Bien Inmueble.
id_shp_p varchar(30) Este campo relaciona esta tabla con la tabla geo_predio
  fecha_levantamiento_p varchar(15) Fecha aproximada en la que se realiza el levantamiento Topográfico.
id_estado_fk integer Llave foránea que relaciona esta tabla con la tabla estado.
  fecha_creacion_p date DEFAULT now() Este Campo almacena la fecha en que crea el registro
  fecha_modifica_p date Este campo almacena la ultima fecha en que se modifica el registro.
  migracion_siga date Este campo almacena la fecha en que el registro fue reportado o cargado en el aplicativo SIGA (Sistema de Información y Gestión de Activos, CISA -Central de Inversiones S.A.)
id_capa integer Llave foránea que relaciona esta tabla con la tabla Capa
  doc_calidad_bien varchar(150) Este campo almacena la ruta donde está almacenado en el Servidor, el documento en pdf del último Concepto de Calidad del Bien.(Nuevo Formato aprobado MAJA01.03.03.18.P14.F02 y MAJA01.03.03.18.P14.F03 del 13 de Junio/2018)
  fecha_expedicion_cb_p date Almacena la fecha de expedición del documento "Respuesta Calidad del Bien"
  orfeo_cb_p varchar(30) Este campo muestra el Número de Orfeo con el que se haya dado un Concepto de Calidad del Bien.
  documento_p varchar(150) Este campo almacena la ruta en el servidor, donde está guardado el expediente físico escaneado en formato PDF
  foto_p varchar(150) Este campo almacena la ruta en el servidor, donde esta guardada la fotografía del bien inmueble.
  ubica_archivo_p varchar(60) Ubicación donde se encuentra guardado el expediente o carpetas dentro del archivo físico. (en el archivador)
  mensaje_p varchar(120) En este campo se almacenan algunos mensajes que se mostraran en el Geovisor de la IDESC
